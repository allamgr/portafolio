# Components Based

For those ones who already have a template and just want to use any component separated, you can just import that into your project without any problem.

```js
# App.js
import { PersonalInfo  } from '@allamgr/portafolio'

function App() {
  let personalInfoData = {
    name: 'Allam Galan Rosa',
    position: 'Full Stack Developer',
    brief: `Experienced Software Developer with more than 8 years. `,
    email: 'agrtest',
    emailDomain: 'gmail.com',
    location: 'Dom. Rep.',
    phone: '859-555-5555',
    gender: 'Male',
    github: 'mygithub',
    linkedin: 'mylinkedin',
    twitter: 'mytwitter',
  }
  return (
    <div>
      <PersonalInfo {...personalInfoData}/>
    </div>
  );
}

export default App;
```
